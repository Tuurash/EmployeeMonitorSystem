using System.Text;

using System.Windows.Forms;
using System.Data;

using CourierServiceManagement.DatabaseAccessLayer;


namespace CourierServiceManagement.BusinessLayer
{
    class NewIdRepository
    {
        private Connection con { get; set; }
        private DataSet Ds { get; set; }

        private int Id { get; set; }
        private string Name { get; set; }
        private int PhoneNumber { get; set; }
        private string Designation { get; set; }
        private string Password { get; set; }
        private DateTime JoiningDate { get; set; }
        private string Address { get; set; }

        internal NewIdRepository()
        {
            this.con = new Connection();
        }

        internal NewIdRepository(int id,string name,int phonenumber,string designation,string password,DateTime joiningDate,string address)
        {
            this.Id = id;
            this.Name = name;
            this.PhoneNumber = phonenumber;
            this.Designation = designation;
            this.Password = password;
            this.JoiningDate = joiningDate;
            this.Address = address;
        }

        internal NewIdRepository(int id, string name, int phonenumber, string designation,DateTime joiningdate,string address)
        {
            this.Id = id;
            this.Name = name;
            this.PhoneNumber = phonenumber;
            this.Designation = designation;
            this.JoiningDate = joiningdate;
            this.Address = address;

        }

        public DataSet DisplayNewId(string sql = "select * from NewId;")
        {
            this.Ds = this.con.ExecuteQuery(sql);
            return this.Ds;
        }

        public void Insert()
        {
           
                string sql = @"insert into NewId
                values ('" + this.Name + "','" + this.Address + "'," + this.PhoneNumber + ", '" + this.Designation + "', '" + this.Password + "', '" + this.JoiningDate + @"');";
                try
                {
                    this.con.ExecuteUpdateQuery(sql);
                    MessageBox.Show("Insertion Done.");

                }
                catch (Exception exc)
                {
                    MessageBox.Show("Error: " + exc.Message);
                }
            }


        }
}
